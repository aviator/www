def body
  @body ||= if raw_body.length > 0
    if Aviator::Compatibility::RUBY_1_8_MODE
      clean_body = raw_body.gsub(/\ /, ' ')
    else
      clean_body = raw_body
    end

    Hashish.new(JSON.parse(clean_body))
  else
    Hashish.new({})
  end
end
